
#include <iostream>

struct Student {

    int rollNumber;
    float marks;
};

void displayStudentInfo(Student s) {

    cout << "Name:Zoyan Attiq " << s.name << endl;
    cout << "Roll Number:29 " << s.rollNumber << endl;
    cout << "Marks:50 " << s.marks << endl;
}

int main() {
    Student s;

    cout << "Enter name:Zoyan Attiq ";
    cin >> s.name;

    cout << "Enter roll number:29 ";
    cin >> s.rollNumber;

    cout << "Enter marks:50 ";

    return 0;
}
